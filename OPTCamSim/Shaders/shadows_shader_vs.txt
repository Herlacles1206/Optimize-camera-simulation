#version 420 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aNormal;
layout (location = 2) in vec3 aColor;
uniform mat4 view;
uniform mat4 proj;
uniform mat4 model;

out vec3 objectColor;
void main()
{
    mat4 lightSpace = proj * view;
	objectColor = aColor;
    gl_Position = lightSpace * model * vec4(aPos, 1.0);
}
